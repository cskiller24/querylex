--- FINAL SQL ---
SELECT
  d.dept_no,
  d.dept_name,
  ranked.emp_no,
  ranked.first_name,
  ranked.last_name,
  ranked.salary
FROM (
  SELECT
    e.emp_no,
    e.first_name,
    e.last_name,
    de.dept_no,
    s.salary,
    ROW_NUMBER() OVER (PARTITION BY de.dept_no ORDER BY s.salary DESC) AS rn
  FROM employees e
  INNER JOIN dept_emp de ON e.emp_no = de.emp_no
    AND de.to_date = '9999-01-01'
  INNER JOIN salaries s ON e.emp_no = s.emp_no
    AND s.to_date = '9999-01-01'
) ranked
INNER JOIN departments d ON ranked.dept_no = d.dept_no
WHERE ranked.rn = 1
ORDER BY d.dept_no;
