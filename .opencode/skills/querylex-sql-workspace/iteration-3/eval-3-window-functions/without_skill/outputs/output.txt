-- Task 2: Highest paid employee in each department
-- Uses ROW_NUMBER() window function partitioned by department, ordered by salary DESC.
-- Only current salary and current department assignments are considered.

WITH ranked AS (
  SELECT
    e.first_name,
    e.last_name,
    d.dept_name,
    s.salary,
    ROW_NUMBER() OVER (PARTITION BY de.dept_no ORDER BY s.salary DESC) AS rn
  FROM employees e
  JOIN salaries s ON e.emp_no = s.emp_no AND s.to_date = '9999-01-01'
  JOIN dept_emp de ON e.emp_no = de.emp_no AND de.to_date = '9999-01-01'
  JOIN departments d ON de.dept_no = d.dept_no
)
SELECT first_name, last_name, dept_name, salary
FROM ranked
WHERE rn = 1
ORDER BY dept_name;
