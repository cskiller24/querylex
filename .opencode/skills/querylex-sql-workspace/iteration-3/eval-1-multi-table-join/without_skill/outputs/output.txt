SELECT
    e.emp_no,
    e.first_name,
    e.last_name,
    e.hire_date,
    s.salary,
    d.dept_name
FROM employees e
JOIN salaries s ON e.emp_no = s.emp_no AND s.to_date = '9999-01-01'
JOIN dept_emp de ON e.emp_no = de.emp_no AND de.to_date = '9999-01-01'
JOIN departments d ON de.dept_no = d.dept_no
WHERE e.hire_date >= DATE_SUB(CURDATE(), INTERVAL 5 YEAR)
ORDER BY e.hire_date DESC;

-- Three inner joins: employeesâ†’salaries (current salary via to_date='9999-01-01'),
-- employeesâ†’dept_emp (current dept via to_date='9999-01-01'),
-- dept_empâ†’departments (dept name).
-- Filter: hire_date within last 5 years using DATE_SUB(CURDATE(), INTERVAL 5 YEAR).
-- Results ordered by most recent hire first.
