-- Show all employees hired in the last 5 years with their current salary and department
SELECT 
    e.emp_no,
    e.first_name,
    e.last_name,
    e.hire_date,
    s.salary AS current_salary,
    d.dept_name AS department
FROM employees e
INNER JOIN salaries s ON e.emp_no = s.emp_no AND s.to_date = '9999-01-01'
INNER JOIN dept_emp de ON e.emp_no = de.emp_no AND de.to_date = '9999-01-01'
INNER JOIN departments d ON de.dept_no = d.dept_no
WHERE e.hire_date >= DATE_SUB(CURDATE(), INTERVAL 5 YEAR)
ORDER BY e.hire_date DESC;

-- Explanation:
-- 1. Filter employees by hire_date >= 5 years ago using DATE_SUB with CURDATE()
-- 2. Join salaries filtered to current rows (to_date = '9999-01-01') for current salary
-- 3. Join dept_emp filtered to current rows (to_date = '9999-01-01') for current department assignment
-- 4. Join departments to get the human-readable department name
-- 5. ORDER BY hire_date DESC so most recently hired appear first
