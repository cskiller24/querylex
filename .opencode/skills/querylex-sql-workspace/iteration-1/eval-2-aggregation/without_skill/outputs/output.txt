SELECT
    d.dept_name,
    ROUND(AVG(s.salary), 2) AS avg_salary
FROM departments d
JOIN dept_emp de ON d.dept_no = de.dept_no AND de.to_date = '9999-01-01'
JOIN salaries s ON de.emp_no = s.emp_no AND s.to_date = '9999-01-01'
GROUP BY d.dept_no, d.dept_name
ORDER BY avg_salary DESC;

-- Approach:
-- 1. Start from departments as the dimension table (gives us department names).
-- 2. Join dept_emp with to_date = '9999-01-01' to get only current department
--    assignments (historical assignments are ignored).
-- 3. Join salaries also with to_date = '9999-01-01' to get only each employee's
--    current salary, avoiding duplicate/historical salary rows.
-- 4. GROUP BY both dept_no and dept_name (dept_no ensures uniqueness since
--    name alone might not be a unique key; dept_name is needed for display).
-- 5. AVG(s.salary) computes the per-department average, ROUND(..., 2) for
--    clean currency formatting.
-- 6. ORDER BY avg_salary DESC so the highest-paying department appears first.
--
-- Note: This counts each employee's current salary only once, in their current
-- department only. An employee assigned to multiple departments historically
-- will only appear in their current (to_date = '9999-01-01') department.
