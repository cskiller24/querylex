- LEFT JOIN dept_emp (current employees: de.to_date = '9999-01-01') to count employees per department

Generated SQL:
SELECT e.first_name, e.last_name, d.dept_name, COUNT(de.emp_no) AS employee_count
FROM dept_manager dm
JOIN employees e ON dm.emp_no = e.emp_no
JOIN departments d ON dm.dept_no = d.dept_no
LEFT JOIN dept_emp de ON dm.dept_no = de.dept_no AND de.to_date = '9999-01-01'
WHERE dm.to_date = '9999-01-01'
GROUP BY dm.emp_no, dm.dept_no, e.first_name, e.last_name, d.dept_name
ORDER BY employee_count DESC
