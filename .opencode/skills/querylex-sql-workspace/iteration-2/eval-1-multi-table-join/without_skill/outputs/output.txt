SELECT
    e.emp_no,
    e.first_name,
    e.last_name,
    e.hire_date,
    s.salary AS current_salary,
    d.dept_name AS department
FROM employees e
LEFT JOIN salaries s
    ON e.emp_no = s.emp_no
    AND s.to_date = '9999-01-01'
LEFT JOIN dept_emp de
    ON e.emp_no = de.emp_no
    AND de.to_date = '9999-01-01'
LEFT JOIN departments d
    ON de.dept_no = d.dept_no
WHERE e.hire_date >= DATE_SUB(CURDATE(), INTERVAL 5 YEAR)
ORDER BY e.hire_date DESC;

-- Explanation:
-- 1. Filter employees hired in the last 5 years via WHERE on hire_date.
-- 2. LEFT JOIN salaries with to_date = '9999-01-01' to get the current salary.
-- 3. LEFT JOIN dept_emp with to_date = '9999-01-01' to get the current department assignment.
-- 4. LEFT JOIN departments to resolve dept_no to the human-readable dept_name.
-- LEFT JOINs are used so employees without a current salary or dept record are
-- still included (salary/dept will be NULL), matching the "Show me ALL employees" intent.
