SELECT e.emp_no, e.first_name, e.last_name, e.hire_date,
       s.salary, d.dept_name
FROM employees e
JOIN salaries s ON e.emp_no = s.emp_no AND s.to_date = '9999-01-01'
JOIN dept_emp de ON e.emp_no = de.emp_no AND de.to_date = '9999-01-01'
JOIN departments d ON de.dept_no = d.dept_no
WHERE e.hire_date >= DATE_SUB(CURRENT_DATE, INTERVAL 5 YEAR);

-- Explanation:
-- Joins employees to salaries (current salary via to_date = '9999-01-01'),
-- dept_emp (current department via to_date = '9999-01-01'), and departments.
-- Filters to employees hired within the last 5 years relative to today.
