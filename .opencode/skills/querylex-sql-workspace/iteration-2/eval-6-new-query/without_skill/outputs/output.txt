SELECT 
    dm.emp_no AS manager_emp_no,
    e.first_name,
    e.last_name,
    COUNT(DISTINCT de.emp_no) AS employee_count
FROM dept_manager dm
JOIN dept_emp de ON dm.dept_no = de.dept_no 
    AND de.to_date = '9999-01-01'
JOIN employees e ON dm.emp_no = e.emp_no
WHERE dm.to_date = '9999-01-01'
GROUP BY dm.emp_no, e.first_name, e.last_name
ORDER BY employee_count DESC;

-- Explanation:
-- Join current managers (dept_manager WHERE to_date = '9999-01-01') with 
-- current department employees (dept_emp WHERE to_date = '9999-01-01') on dept_no.
-- COUNT(DISTINCT de.emp_no) handles employees in multiple departments under the same 
-- manager, and managers who oversee multiple departments (aggregated per manager).
-- Includes manager name from employees table for readability.
