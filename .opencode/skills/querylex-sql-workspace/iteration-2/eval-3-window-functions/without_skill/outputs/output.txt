SELECT e.emp_no, e.first_name, e.last_name,
       d.dept_no, d.dept_name, s.salary
FROM employees e
JOIN salaries s ON e.emp_no = s.emp_no AND s.to_date = '9999-01-01'
JOIN dept_emp de ON e.emp_no = de.emp_no AND de.to_date = '9999-01-01'
JOIN departments d ON de.dept_no = d.dept_no
WHERE (de.dept_no, s.salary) IN (
  SELECT de2.dept_no, MAX(s2.salary)
  FROM salaries s2
  JOIN dept_emp de2 ON s2.emp_no = de2.emp_no
  WHERE s2.to_date = '9999-01-01' AND de2.to_date = '9999-01-01'
  GROUP BY de2.dept_no
)
ORDER BY d.dept_no;

-- Explanation: The subquery finds the max current salary per department by
-- joining salaries and dept_emp filtered to current rows (to_date='9999-01-01').
-- The outer query joins back to get the employee details for those (dept_no, salary)
-- pairs. If multiple employees share the same max salary in a department, ALL are
-- returned. Unused departments (no current employees) are excluded.
